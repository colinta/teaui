# Checkbox

A toggle checkbox with label. Toggles on click or Enter key.

<TerminalScreenshot name="checkbox" title="Checkbox" />

## React Example

```tsx
import React, {useState} from 'react'
import {Stack, Checkbox, run} from '@teaui/react'

function App() {
  const [dark, setDark] = useState(false)
  const [auto, setAuto] = useState(true)

  return (
    <Stack.down>
      <Checkbox title="Dark mode" value={dark} onChange={setDark} />
      <Checkbox title="Auto-save" value={auto} onChange={setAuto} />
    </Stack.down>
  )
}

run(<App />)
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `title` | `string` | — | Label text |
| `value` | `boolean` | `false` | Checked state |
| `onChange` | `(value: boolean) => void` | — | Toggle callback |
| `hotKey` | `HotKey` | — | Keyboard shortcut |
